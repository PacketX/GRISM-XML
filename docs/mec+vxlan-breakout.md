## Config XML

```xml
<configSet reboot="no">
    <args>
        <s1cCorrelation>True</s1cCorrelation>
        <flowExtensionGtpTunnelhdr>True</flowExtensionGtpTunnelhdr>
        <vxlanCorrelation>True</vxlanCorrelation>
        <vxlanCorrelationPort>P8</vxlanCorrelationPort>   
    </args>
    <filters>
        <in-tunnels>
            <GTP>True</GTP>     
            <VXLAN>True</VXLAN>
        </in-tunnels>
    </filters>
</configSet>
```
